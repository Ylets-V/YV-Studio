    <script>
        const audio = document.getElementById('audioSource');

        function openSidebar(id) { 
            closeSidebar();
            document.getElementById(id).classList.add('active'); 
        }
        function closeSidebar() { 
            document.querySelectorAll('.sidebar').forEach(s => s.classList.remove('active'));

        function toggleVisibility(id) {
            const el = document.getElementById(id);
            const btn = document.getElementById('btn-' + id);
            const isHidden = el.style.display === 'none';
            el.style.display = isHidden ? '' : 'none';
            btn.classList.toggle('active', isHidden);
        }

        function applyTheme(theme, btn) {
            document.querySelectorAll('.swatch-circle').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
            const root = document.documentElement;
            const themes = {
                dark: { bg: '#000', op: 0.7 },
                light: { bg: '#f1f5f9', op: 0.4 },
                green: { bg: '#064e3b', op: 0.6 }
            };
            root.style.setProperty('--theme-bg', themes[theme].bg);
            document.getElementById('opLeft').value = themes[theme].op;
            document.getElementById('opRight').value = themes[theme].op;
            updatePanelOpacity();
        }

        function applyCustomColor(color) {
            document.querySelectorAll('.swatch-circle').forEach(b => b.classList.remove('active'));
            document.getElementById('customSwatch').classList.add('active');
            document.documentElement.style.setProperty('--theme-bg', color);
        }

        function updatePanelOpacity() {
            const opL = document.getElementById('opLeft').value;
            const opR = document.getElementById('opRight').value;
            document.documentElement.style.setProperty('--panel-bg-left', `rgba(0, 0, 0, ${opL})`);
            document.documentElement.style.setProperty('--panel-bg-right', `rgba(0, 0, 0, ${opR})`);
        }

        function handleCustomFontFile(e) {
            const file = e.target.files[0];
            if(!file) return;
            const reader = new FileReader();
            reader.onload = (event) => {
                const fontName = 'UserFont_' + Date.now();
                const fontFace = new FontFace(fontName, event.target.result);
                fontFace.load().then(loadedFace => {
                    document.fonts.add(loadedFace);
                    document.getElementById('manualFontT').value = fontName;
                    updateUI();
                });
            };
            reader.readAsArrayBuffer(file);
        }

        function updateWebAdjustments() {
            document.documentElement.style.setProperty('--bg-brightness', document.getElementById('bgBrightness').value);
            document.documentElement.style.setProperty('--bg-blur', document.getElementById('bgBlur').value + 'px');
        }

        function updatePinFilters() {
            document.documentElement.style.setProperty('--pin-brightness', document.getElementById('pinBrightness').value);
            document.documentElement.style.setProperty('--pin-contrast', document.getElementById('pinContrast').value);
            document.documentElement.style.setProperty('--pin-saturate', document.getElementById('pinSaturate').value);
        }

        function updateDimensions() {
            const w = document.getElementById('inWidth').value || 380;
            const h = document.getElementById('inHeight').value || 675;
            const scale = document.getElementById('uiScale').value;
            document.documentElement.style.setProperty('--pin-width', w + 'px');
            document.documentElement.style.setProperty('--pin-height', h + 'px');
            document.documentElement.style.setProperty('--pin-scale', scale);
        }

        function updateUI() {
            const outT = document.getElementById('outTitle');
            const outA = document.getElementById('outArtist');

            if(manualT) outT.style.fontFamily = manualT + ', sans-serif';
            
            outA.innerText = document.getElementById('inArtist').value;
            outA.style.fontSize = document.getElementById('tsArtist').value + 'px';

            if(manualA) outA.style.fontFamily = manualA + ', sans-serif';

            document.documentElement.style.setProperty('--wave-scale', document.getElementById('waveScale').value);
            document.documentElement.style.setProperty('--color-waves', document.getElementById('cWaves').value);
            document.documentElement.style.setProperty('--color-player', document.getElementById('cPlayer').value);
            document.documentElement.style.setProperty('--glass-blur', document.getElementById('inBlurPx').value + 'px');
            document.getElementById('playBtn').style.color = document.getElementById('cPlayer').value;
        }

        function handleFile(e, type) {
            const file = e.target.files[0];
            if(!file) return;
            const url = URL.createObjectURL(file);
            if(type === 'audio') {
                audio.src = url;
                audio.onloadedmetadata = () => document.getElementById('durTime').innerText = formatTime(audio.duration);
            }
            if(type === 'pin') document.getElementById('pinFrame').style.backgroundImage = `url(${url})`;
            if(type === 'web') document.getElementById('webBackground').style.backgroundImage = `url(${url})`;
        }

        function togglePlay(e) {
            if(e && e.stopPropagation) e.stopPropagation();
            if(!audio.src) return;
            if(!audioCtx) {
                audioCtx = new (window.AudioContext || window.webkitAudioContext)();
                analyser = audioCtx.createAnalyser();
                const source = audioCtx.createMediaElementSource(audio);
                source.connect(analyser);
                analyser.connect(audioCtx.destination);
                analyser.fftSize = 256;
                dataArray = new Uint8Array(analyser.frequencyBinCount);
            }
            if(audio.paused) {
                audio.play();

        function formatTime(s) { const m=Math.floor(s/60), sc=Math.floor(s%60); return `${m.toString().padStart(2,'0')}:${sc.toString().padStart(2,'0')}`; }

        function draw() {
            requestAnimationFrame(draw);
            if(analyser) analyser.getByteFrequencyData(dataArray);
            
            const w = canvas.width = canvas.clientWidth * 2;
            const h = canvas.height = canvas.clientHeight * 2;
            ctx.clearRect(0, 0, w, h);
            
            const waveColor = document.getElementById('cWaves').value;
            const waveScaleFactor = parseFloat(document.documentElement.style.getPropertyValue('--wave-scale') || 1);
            const time = Date.now() * 0.003;
            
            let bass = 0; 
            if(dataArray && !audio.paused) { 
                for(let i=0; i<5; i++) bass += dataArray[i]; 
                bass = (bass/5)/255; 
            }
            
            document.documentElement.style.setProperty('--player-scale', 1 + (bass * 0.15));

            const drawWave = (amp, freq, speed, op, thick) => {
                const scaledAmp = amp * waveScaleFactor;
                ctx.beginPath(); 
                ctx.lineWidth = thick; 
                ctx.strokeStyle = waveColor + op;
                for(let x=0; x<=w; x+=4) {
                    const y = (h/2) + Math.sin(x*freq + time*speed) * scaledAmp;
                    if(x===0) ctx.moveTo(x,y); else ctx.lineTo(x,y);
                }
                ctx.stroke();
            };

            drawWave(25+(bass*60), 0.004, 0.5, "33", 10);
            drawWave(15+(bass*80), 0.008, 1.2, "88", 6);
            drawWave(10+(bass*30), 0.015, 2.0, "FF", 3);
        }

        window.onload = draw;

        async function exportMp4() {
            if(isExporting) return;
            const btn = document.getElementById('btnMp4');
            const status = document.getElementById('exportStatus');
            
            if (!audio.src) {
                status.innerText = "Error: Carga un audio primero";
                status.classList.add('active');
                setTimeout(() => status.classList.remove('active'), 2000);
                return;
            }

            try {
                isExporting = true;
                btn.disabled = true;
                status.innerText = "CAPTURA DE VIDEO (10s)...";
                status.classList.add('active');

                // 1. Capturar fondo estático
                const pin = document.getElementById('pinFrame');
                const originalScale = document.documentElement.style.getPropertyValue('--pin-scale');
                document.documentElement.style.setProperty('--pin-scale', '1');
                await new Promise(r => setTimeout(r, 200));
                
                const captureCanv = await html2canvas(pin, { useCORS: true, scale: 2 });
                const pinData = captureCanv.toDataURL('image/png');

                // 2. Grabar visualizador
                const stream = canvas.captureStream(30);
                const recorder = new MediaRecorder(stream, { 
                    mimeType: 'video/webm;codecs=vp9',
                    videoBitsPerSecond: 3000000 
                });
                
                const chunks = [];
                recorder.ondataavailable = e => { if(e.data.size > 0) chunks.push(e.data); };
                
                recorder.onstop = async () => {
                    status.innerText = "INICIALIZANDO RENDERIZADOR...";
                    
                    try {
                        if (!ffmpeg.isLoaded()) {
                            await ffmpeg.load();
                        }

                        const videoBlob = new Blob(chunks, { type: 'video/webm' });
                        ffmpeg.FS('writeFile', 'vis.webm', await fetchFile(videoBlob));
                        ffmpeg.FS('writeFile', 'bg.png', await fetchFile(pinData));
                        
                        // Cargar audio real
                        const audioRes = await fetch(audio.src);
                        const audioBuf = await audioRes.arrayBuffer();
                        ffmpeg.FS('writeFile', 'audio.mp3', new Uint8Array(audioBuf));

                        status.innerText = "PROCESANDO MP4 FINAL...";

                        // Comando optimizado para un solo hilo
                        await ffmpeg.run(
                            '-loop', '1', '-i', 'bg.png',
                            '-i', 'vis.webm',
                            '-i', 'audio.mp3',
                            '-filter_complex', '[1:v]scale=iw:ih[v1];[0:v][v1]overlay=0:0:shortest=1',
                            '-pix_fmt', 'yuv420p',
                            '-c:v', 'libx264',
                            '-preset', 'ultrafast',
                            '-threads', '1', // Forzamos 1 hilo explícitamente
                            '-c:a', 'aac',
                            '-shortest',
                            'out.mp4'
                        );

                        const data = ffmpeg.FS('readFile', 'out.mp4');
                        const downloadLink = document.createElement('a');
                        downloadLink.href = URL.createObjectURL(new Blob([data.buffer], { type: 'video/mp4' }));
                        downloadLink.download = 'YV_STUDIO_EXPORT.mp4';
                        downloadLink.click();

                        status.innerText = "¡EXPORTACIÓN COMPLETADA!";
                    } catch (e) {
                        console.error(e);
                        status.innerText = "Error de Render (Revisa Consola)";
                    } finally {
                        setTimeout(() => status.classList.remove('active'), 3000);
                        isExporting = false;
                        btn.disabled = false;
                        document.documentElement.style.setProperty('--pin-scale', originalScale);
                    }
                };

                // Iniciar audio y grabación
                if (audio.paused) togglePlay(); 
                audio.currentTime = 0;
                recorder.start();
                
                setTimeout(() => {
                    recorder.stop();
                    audio.pause();
                }, 10000); 

            } catch (err) {
                console.error(err);
                isExporting = false;
                btn.disabled = false;
                status.innerText = "Error en el proceso";
                setTimeout(() => status.classList.remove('active'), 3000);
            }
        }
